import java.util.Arrays;
class Main {
    public static void main(String[] args) {
        int[] a={1,5,6,4,2};
        for (int p=1;p<a.length;p++){
            
            for (int i=0;i<a.length-p;i++){
                
                if(a[i]>a[i+1]){
                    int t=a[i];
                    a[i]=a[i+1];
                    a[i+1]=t;
        }
    
    
    }
}
System.out.print(Arrays.toString(a));
}

}
